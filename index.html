function sayToken(tok){ // e.g., "O3" -> "O three" const m = tok.match(/^([A-Z])(\d)$/); if (!m) return tok; const num = ({1:'one',2:'two',3:'three'})[m[2]] || m[2]; return `${m[1]} ${num}`; } function toUnitPhrase(magnitude, dirWord) { // dirWord is one of: above, below, north, south, east, west, northeast, northwest, southeast, southwest const isVertical = (dirWord === 'above' || dirWord === 'below'); const unitWord = isVertical ? 'level' : 'cell'; const magWord = ({'1':'one','2':'two','3':'three'})[String(magnitude)] || String(magnitude); const plural = (magWord === 'one') ? '' : 's'; // Speak diagonals as two words (“north east”), clearer in TTS const spokenDir = dirWord.replace(/northwest/i,'north west') .replace(/northeast/i,'north east') .replace(/southwest/i,'south west') .replace(/southeast/i,'south east'); return `${magWord} ${unitWord}${plural} ${spokenDir}`; } function verbalizeSentence(sentence){ // Input looks like: "O3 is 3 south of P9 and J7 is 1 above of O3 and P9 is 2 northeast of J7" // We’ll convert ONLY the relation phrase " " into " cell(s)/level(s) " // and read tokens like "O3" as "O three". const parts = sentence.split(/\s+and\s+/i).map(p => p.trim()); return parts.map(part => { // 1) Expand grid tokens (O3 -> O three) let spoken = part.replace(/([A-Z]\d)/g, m => sayToken(m)); // 2) Inject units into the relation phrase " " // Capture: magnitude + direction word spoken = spoken.replace( /\b(1|2|3)\s+(above|below|north(?:east|west)?|south(?:east|west)?|east|west)\b/i, (_, mag, dir) => toUnitPhrase(mag, dir.toLowerCase()) ); // 3) Ensure preposition is natural (“above”/“below” already ok; diagonals kept as two words) // (Nothing else needed; we’ve already normalized diagonals inside toUnitPhrase.) return spoken; }); }
